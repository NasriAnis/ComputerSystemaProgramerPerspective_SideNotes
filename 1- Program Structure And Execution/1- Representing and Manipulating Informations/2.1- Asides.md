# About the <stdint.h>, <inttypes.h> and <limits.h> C library

```c
#include <inttypes.h>
#include <stdint.h>

#include <limits.h>

#include <stdio.h>

int main() {
  printf("int 32 min :          %d\n", INT32_MIN);
  printf("int 32 max :          %d\n", INT32_MAX);
  printf("unsigned int 32 man : %u\n", UINT32_MAX);
  printf("int 64 min :          %ld\n", INT64_MIN);
  printf("int 64 max :          %ld\n", INT64_MAX);
  printf("unsigned int 64 max : %lu\n", UINT64_MAX);

  printf("\n");

  printf("int 32 min :          %" PRId32 "\n", INT32_MIN);
  printf("int 32 max :          %" PRId32 "\n", INT32_MAX);
  printf("unsigned int 32 man : %" PRIu32 "\n", UINT32_MAX);
  printf("int 64 min :          %" PRId64 "\n", INT64_MIN);
  printf("int 64 max :          %" PRId64 "\n", INT64_MAX);
  printf("unsigned int 64 max : %" PRIu64 "\n", UINT64_MAX);

  printf("\n");

  printf("int max :             %d\n", INT_MAX);
  printf("int min :             %d\n", INT_MIN);
  printf("unsigned int max :    %lu\n", UINT_MAX);
  return 0;
}
```
